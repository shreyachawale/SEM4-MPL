DATA_SEG SEGMENT
    STRING DB 'accba'
    LEN DW 0005H
DATA_SEG ENDS

CODE_SEG SEGMENT
    ASSUME CS: CODE_SEG, DS:DATA_SEG

START:MOV AX,DATA_SEG
      MOV DS,AX

      LEA SI,STRING
      MOV DI,SI
      MOV DI,LEN
      DEC DI
      MOV CX,LEN

BACK:MOV AL,DS:[SI]
     CMP AL,DS:[DI]
     JNZ NOT_PALINDROME
     INC SI
     DEC DI
     LOOP BACK 
     MOV BX,0AAAAH

NOT_PALINDROME: MOV BX,0FFFFH

CODE_SEG ENDS
END START
