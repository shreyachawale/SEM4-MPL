DATA_SEG SEGMENT
     source_arr DB 28H,11H,22H,33H,44H
     dest_arr DB 5 DUP(?)
DATA_SEG ENDS

CODE_SEG SEGMENT
    ASSUME CS: CODE_SEG, DS: DATA_SEG
START: MOV AX,DATA_SEG
       MOV DS,AX

       LEA SI,source_arr
       LEA DI,dest_arr
       MOV CX,5
NEXT:
       MOV AL,[SI]
       MOV [DI],AL
       INC SI
       INC DI
       LOOP NEXT
 
       MOV AH,4CH
       INT 21H

CODE_SEG ENDS
END START
