DATA_SEG SEGMENT
     ARR1 DB 10H,20H,30H,40H,50H
     ARR2 DB 5 DUP(?)
DATA_SEG ENDS

CODE_SEG SEGMENT
    ASSUME CS: CODE_SEG, DS: DATA_SEG
START: MOV AX,DATA_SEG
       MOV DS,AX

       MOV ES,AX
       MOV CL,08
NEXT:
       MOV SI, OFFSET ARR1
       MOV DI, OFFSET ARR2
       MV AL, DS:[SI]
       MOV ES:[DI],AL
       INC SI
       INC DI
       LOOP BACK
 
       MOV AH,4CH
       INT 21H

CODE_SEG ENDS
END START
