DATA_SEG SEGMENT
    NUM DB 06H
    ONES DB 0
    ZEROS DB 0
DATA_SEG ENDS

CODE_SEG SEGMENT
     ASSUME CS: CODE_SEG, DS: DATA_SEG

START:MOV AX,DATA_SEG
      MOV DS,AX
      MOV AL,NUM
      MOV CX,06

BACK:ROR AL,1
     JC INC_ONES
     INC ZEROS
     JMP EXIT

INC_ONES: INC_ONES

EXIT: LOOP BACK

      MOV AH,4CH
      INT 21H

CODE_SEG ENDS
END START
