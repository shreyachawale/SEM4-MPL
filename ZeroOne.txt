DATA_SEG SEGMENT
    NUM DB 06H
DATA_SEG ENDS

CODE_SEG SEGMENT
     ASSUME CS: CODE_SEG, DS: DATA_SEG

START:MOV AX,DATA_SEG
      MOV DS,AX
      MOV BH,00H
      MOV BL,00H
      MOV AL,NUM
      MOV CX,08

BACK:ROR AL,1
     JC INC_ONES
     INC BH
     JMP EXIT

INC_ONES: INC BL

EXIT: LOOP BACK

      MOV AH,4CH
      INT 21H

CODE_SEG ENDS
END START
