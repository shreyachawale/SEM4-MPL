DATA_SEG SEGMENT
    ARR DB 11H,12H,13H,14H,15H,16H
    EVEN_CNT DB 0
    ODD_CNT DB 0
DATA_SEG ENDS

CODE_SEG SEGMENT
     ASSUME CS: CODE_SEG, DS: DATA_SEG

START:MOV AX,DATA_SEG
      MOV DS,AX

      MOV SI,OFFSET ARR
      MOV CL,10

BACK:MOV AL,[SI]
     ROR AL,1
     JC INC_ODD
     INC EVEN_CNT
     JMP FOR_SI_INC

INC_ODD: INC ODD_CNT
         JMP FOR_SI_INC

FOR_SI_INC: INC SI
            LOOP BACK 

            MOV AH,4CH
            INT 21H


CODE_SEG ENDS
END START
